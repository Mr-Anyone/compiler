function is_equal gives int[
    int a,
    int b,
]{
    ret (a eq b) + (a ne b) + (a le b) + (a lt b) + (a ge b) + (a gt b);
}

function fib gives int[
    int a,
]{
    if a le 1 then 
        ret a;
    end

    ret fib(a-1 ,) + fib(a-2,);
}
